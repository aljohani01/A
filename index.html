const express = require('express'); const bodyParser = require('body-parser'); const fs = require('fs'); const app = express(); const PORT = 3000; app.use(bodyParser.json()); // مسار لتحميل البيانات app.get('/kashoosh/data', (req, res) => { try { const data = fs.readFileSync('data.json'); res.json(JSON.parse(data)); } catch (error) { res.status(500).json({ error: 'فشل في تحميل البيانات' }); } }); // مسار لحفظ البيانات app.post('/kashoosh/data', (req, res) => { try { fs.writeFileSync('data.json', JSON.stringify(req.body)); res.json({ success: true }); } catch (error) { res.status(500).json({ error: 'فشل في حفظ البيانات' }); } }); app.listen(PORT, () => { console.log(`الخادم يعمل على المنفذ ${PORT}`); });
